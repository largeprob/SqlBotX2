# 普通指标

```JSON
{
    "metrics": [
        {
            "name": "新增客户总数",
            "type": "普通指标",
            "expression": "Count(*)",
            "alias": "指标别名",
            "dependencies": [
                {
                    "time": "t1",
                }
            ],
         }
    ],

    "time_ranges": [
        {
            "name": "时间",
            "alias": "t1",

            "tag": "时间日期",
            "operator": "区间",

                "2026-01-01","2026-12-31"
```

"2025年新增客户同比以及2026新增总数是多少？"
```JSON
{
    // 指标用以直接对外展示的指标，包含普通指标和复合指标
    "main_metrics": [
        {
            "name": "2025年新增客户同比",
            "type": "虚拟指标",
            "expression": "(c1-c2)/c2 × 100%",
            "alias": "r1",
         },
        {
            "name": "2026年新增客户总数",
            "type": "普通指标",
            "expression": "Count(f1)",
            "alias": "r2",
            "dependencies":  {
                "time": "t3",
            }
         },
    ],
     //复合指标拆分出来子指标用以计算复合指标的值
    "children_metrics":[
        {
            "name": "2025年新增客户总数",
            "alias": "c1",
            "expression": "COUNT(f1)",
            "dependencies": {
                "time": "t1",
             }
         },
         {
            "name": "2024年新增客户总数",
            "alias": "c2",
            "expression": "COUNT(f1)",
            "dependencies": {
                "time": "t2",
             }
         },
    ],
    //指标用以度量的原子字段
    "metric_fileds":[
        {
            "name": "客户主键",
            "column_name": "Id",
            "alias": "f1",
        },
    ],
    "time_ranges": [
        {
            "name": "时间",
            "alias": "t1",
            "type": "隐式",
            "tag": "时间日期",
            "operator": "区间",
            "value": [
                "2025-01-01","2025-12-31"
            ],
            "result_name":"cte1",
        },
        {
            "name": "时间",
            "alias": "t2",
            "type": "隐式",
            "tag": "时间日期",
            "operator": "区间",
            "value": [
                "2024-01-01","2024-12-31"
            ],
             "result_name":"cte2",
        },
         {
            "name": "时间",
            "alias": "t3",
            "type": "隐式",
            "tag": "时间日期",
            "operator": "区间",
            "value": [
                "2026-01-01","2026-12-31"
            ],
            "result_name":"cte3",
        }
    ],
    "dependent_dims": [
        "年",
    ],
}
```
```SQL
with  ct1 as (select 2025 as year,130 as c1),
ct2 as (select 2024 as year,100 as c2),
ct3 as (select 2026 as year,1001 as r2)
select  FORMAT(((ct1.c1 - ct2.c2) * 1.0 / ct2.c2 * 100), 'N2') + '%' AS '2025年新增客户同比',ct3.r2 as '2026年新增客户总数'
from ct1,ct2,ct3
```
